```sh
# Generated via 
grep -r incept scripts/ | grep '\-\-name\s.*\s' | grep '\-\-alias\s.*\s' | sed 's/.*--name \([A-Za-z0-9\-]\+\) .*--alias \([A-Za-z0-9\-]\+\) .*/\1,\2/g' | grep -v registry | sed 's/\(.*\),\(.*\)/kli export --name \1 --alias \2 > \1_\2.export/' > to_export.sh
```
